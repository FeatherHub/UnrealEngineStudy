UWorld::Tick()
{
  this::BroadcastTickDispatch();       // 프레임 앞: 수신
  // gameplay 
  this::BroadcastTickFlush();          // 프레임 끝: 송신 — 각 UNetDriver::TickFlush()
}

UNetDriver::TickFlush()
{
  this::ServerReplicateActors()
  {
    this::ServerReplicateActors_PrepConnections();     // connection 별 viewer 확정
    this::ServerReplicateActors_BuildConsiderList();   // NetworkObjects 중 NextUpdateTime 도달·비-dormant 만

    for each connection
    this::ServerReplicateActors_PrioritizeActors();    // AActor::IsNetRelevantFor() 필터 + NetPriority × 경과시간 정렬
    this::ServerReplicateActors_ProcessPrioritizedActorsRange()
    {
      // 대역폭 예산 안에서 상위부터:
      Connection:UNetConnection*::CreateChannelByName();  // 첫 복제면 channel open — 클라쪽 spawn 의 근원
      Channel:UActorChannel*::ReplicateActor()
      {
        Replicator:FObjectReplicator::ReplicateProperties(Bunch)
        {
          // ChangelistMgr: 이번 프레임 값 비교는 객체당 1회 (모든 connection 공유)
          this::RepLayout::ReplicateProperties();  // changelist − (이 연결이 ack 한 것) → 비트 기록
        }
        // replicated subobject 도 ReplicationMap 의 replicator 로 동일 반복
        this::SendBunch(&Bunch);        // bunch 가 Connection::SendBuffer 에 패킹됨
      }
    }
  }
  Connection::FlushNet()                // 모인 비트가 packet 이 되어 PacketId 부여
  {
    this::Handler:PacketHandler*::Outgoing();   // 압축 → 암호화
    this::LowLevelSend();                       // socket 송신
  }
}

UNetDriver::TickDispatch()      // UIpNetDriver 는 socket recv 루프
{
  Connection:UNetConnection*::ReceivedRawPacket(Data)
  {
    this::Handler:PacketHandler*::Incoming(); // 복호화·압축해제
                                              //미연결 주소의 packet 은 StatelessConnectHandlerComponent 가 cookie 검증
                                              // 통과 전엔 connection 자체가 없음
    this::ReceivedPacket(Reader)
    {
      // PacketNotify: header 의 seq/ack 처리
      //   내 packet 의 nak → 거기 실렸던 reliable bunch 를 다음 packet 에 재수록 (UChannel::ReceivedNak)
      //   ack → replicator 의 "ack 된 changelist" 갱신
      // packet 내 bunch 순회:
      Channel:UChannel*::ReceivedRawBunch()
      {
        // reliable: channel 별 sequence 로 순서 보장, 결번이면 보류 큐에 대기
        // partial bunch 는 재조립 후 전달
        UActorChannel::ReceivedBunch()
        {
          // open bunch 면: PackageMap:UPackageMapClient* 가 NetGUID→archetype resolve → actor spawn
          Replicator:FObjectReplicator::ReceivedBunch()
          {
            this::RepLayout:FRepLayout*::ReceiveProperties();  // 객체 메모리에 기록 + RepNotify 큐잉
            // RPC 필드면: ReceivedRPC() → 로컬 UFunction 실행
          }
        }
      }
    }
  }
}

고전 스택의 비용 항목

상태 저장   : FObjectReplicator 가 (객체 × 연결) 쌍마다        → 메모리 ∝ N×C
변화 감지   : shadow 와 비교 = UObject 레이아웃 산책 (pull)    → 캐시 비협조, 매 프레임
정책       : IsNetRelevantFor() = 객체당 가상함수 호출         → 배치 불가
직렬화     : bunch 조립이 game thread 에 직렬                  → 병렬화 불가

---

상태를 내부 표현으로 복사(quantize)해 데이터 지향 버퍼에 두고, 
변화는 비교(pull)가 아니라 통지(push)로 받고, 

비교·직렬화 산출물은 connection 들이 공유하며(연결별로는 "무엇을 보냈나" 기록만), 
정책(filter/prioritize)은 config 로 등록되는 시스템으로 만든다.

   Writer(연결당 1개)가 S4 의 마스크를 자기 S5(pending mask)에 OR
   // 연결이 100개여도 75 라는 값의 사본은 늘지 않는다. 마스크만 100벌

⑥ 직렬화 — 연결별 델타
   FReplicationWriter::Write
   // V3(현재) 를 V4(이 연결의 baseline) 대비 SerializeDelta → 비트열.
   // "75" 가 아니라 "옛값에서 75 로의 차분"이 나간다
   // S5 의 마스크를 S6(FReplicationRecord)에 (PacketId 와 함께) 기록
   // pending → in-flight 로 이전 — "실어 보냈고 결과 대기 중"
   여기서 값의 형태:  quantized 값 → 델타 비트열 (연결마다 다를 수 있음)

⑦ 전송 접점
   UReplicationDataStream → UDataStreamChannel 의 bunch → packet
   // 이하는 전송 계층(seq/ack, 암호화·압축)

⑧ 해독 (클라이언트)
   UDataStreamChannel::ReceivedBunch → … → FReplicationReader::Read
   // changemask 를 먼저 읽고 → 마스크가 가리키는 member 만
   // V5(ReceiveStateBuffer, 수신측 baseline) 대비 델타 해독 → V5 를 75(q) 로 갱신
   사본 현황(클라):  V5=75(q) │ V6=옛값 │ V7=옛값

⑨ 적용
   FPropertyReplicationFragment::ApplyReplicatedState()
     // dequantize: V5 의 quantized 값 → external 형식 → V6
     FPropertyReplicationState::PushPropertyReplicationState(Owner, &V7);
     // V6 → V7 (UObject 멤버에 기록)
     FPropertyReplicationState::CallRepNotifies(…);
     // 이전 값과 비교해 OnRep_Health 호출 
     사본 현황(클라):  V5=75(q) │ V6=75 │ V7=75 → gameplay 반응

⑩ 서버, ack 도착 시
   ack  → FReplicationWriter::HandleDeliveredRecord()
          // S6 의 in-flight 해소. V4 baseline 을 75(q) 로 갱신
          // "상대가 아는 과거"가 전진
   nak  → FReplicationWriter::HandleDroppedRecord()
          // 잃어버린 packet 에 실렸던 마스크를 S5(pending)로 되돌림.
          // 다음 기회에 '그때의 최신 V3' 로 다시 직렬화
          // 보냈던 비트의 재전송이 아니라 상태로부터의 재파생
