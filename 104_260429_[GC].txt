class UMyAsyncResource : public UObject 
{
    FRenderResourceHandle GpuHandle;
    char* RawBuffer;

    virtual void BeginDestroy() override {
        Super::BeginDestroy();
        ENQUEUE_RENDER_COMMAND(...)([this](FRHICommandList& RHICmdList) {
            ReleaseGpu(GpuHandle);
        });
        GpuReleaseFence.BeginFence();
    }

    virtual bool IsReadyForFinishDestroy() override {
        return Super::IsReadyForFinishDestroy() && GpuReleaseFence.IsComplete();
    }

    virtual void FinishDestroy() override {
        delete[] RawBuffer;
        Super::FinishDestroy();
    }
};
