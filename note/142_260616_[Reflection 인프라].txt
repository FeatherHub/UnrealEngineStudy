float UMyTestClass::MyFoo(int32 ItemCount, float ItemCost, bool bSharable)
{
    UFunction* Func = FindFunctionChecked(NAME_UMyTestClass_MyFoo);

    if (!Func->GetOwnerClass()->HasAnyClassFlags(CLASS_Native))
    {
        MyTestClass_eventMyUFUNCFoo_Parms Parms;
        Parms.ItemCount = ItemCount;
        Parms.ItemCost  = ItemCost;
        Parms.bSharable = bSharable ? true : false;
        ProcessEvent(Func, &Parms);
        return Parms.ReturnValue;
    }
    else
    {
        return MyUFUNCFoo_Implementation(ItemCount, ItemCost, bSharable);  // 빠른 경로
    }
}

DEFINE_FUNCTION(UMyTestClass::execMyFoo)
{
    P_GET_PROPERTY(FIntProperty, Z_Param_ItemCount);
    P_GET_PROPERTY(FFloatProperty, Z_Param_ItemCost);
    P_GET_UBOOL(Z_Param_bSharable);

    *(float*)Z_Param__Result = 
      P_THIS->MyFoo_Implementation
      (Z_Param_ItemCount, Z_Param_ItemCost, Z_Param_bSharable);
}

핵심은 FindFunction가 UMyTestClass 고정 lookup이 아니라 this->GetClass() 기준이라는 점

BP로 만든 인스턴스라면 GetClass()는 UBlueprintGeneratedClass이고, 
UClass::FindFunctionByName은 먼저 그 class의 FuncMap을 보게 됩니다

그래서 BP override가 있으면 native parent의 MyFoo가 아니라 BP class 쪽 UFunction이 잡힙니다
