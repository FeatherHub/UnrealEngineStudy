void UStruct::SerializeTaggedProperties(
    FArchive& Ar, 
    uint8* Data, 
    UStruct* DefaultsStruct, 
    uint8* Defaults)
{
    if (Ar.IsSaving())
    {
        // --- 저장 경로 ---
        for (FProperty* Prop : ForEachProperty(this))
        {
            // 1. 기본값과 동일하면 생략 (delta serialization)
            if (Prop->Identical(Data, Defaults))
                continue;

            // 2. FPropertyTag 작성: 프로퍼티 메타데이터
            FPropertyTag Tag;
            Tag.Name = Prop->GetFName();     // "Score", "PlayerName"
            Tag.Type = Prop->GetID();        // "IntProperty", "StrProperty"
            Tag.Size = /* 직렬화될 바이트 수 */;
            Ar << Tag;  // 태그를 아카이브에 씀

            // 3. 실제 프로퍼티 값 직렬화
            Prop->SerializeItem(
                FStructuredArchive::FSlot(Ar), 
                Data + Prop->GetOffset_ForInternal()
            );
        }

        // 끝 마커
        FName NoneProp = NAME_None;
        Ar << NoneProp;
    }
}

구조 서술을 분리한다

// "Player가 어떻게 생겼는가"를 딱 한 번만 서술한다
void Describe(Player& p, Handler& h) {
    h.field(p.Score);
    h.field(p.Name);
}

Describe는 아무것도 하지 않는다. 
그냥 "Score가 있고, Name이 있다"고 말할 뿐이다. 
무엇을 할지는 Handler가 결정한다.

struct SaveHandler {
    File& f;
    template<typename T>
    void field(T& value) { f.write(value); }  // 저장
};

struct LoadHandler {
    File& f;
    template<typename T>
    void field(T& value) { f.read(value); }   // 로딩
};

struct MemoryCountHandler {
    size_t total = 0;
    template<typename T>
    void field(T& value) { total += sizeof(value); }  // 계산
};

필드를 추가하면 Describe만 수정한다

FStructuredArchive: field(value) 대신 field("Score", value)
이름이 있으면 순서가 달라져도 괜찮고, JSON으로 출력할 수도 있다.

FPropertyTag + SerializeTaggedProperties
이름을 붙이는 것을 리플렉션이 자동으로 해준다
UPROPERTY가 붙은 필드는 Describe를 직접 쓸 필요조차 없다

FLinkerSave/Load
수백 개의 오브젝트가 서로를 참조하는 그래프 전체를 Describe한다
포인터를 인덱스로 바꾸는 것, NameMap으로 문자열을 압축하는 것
