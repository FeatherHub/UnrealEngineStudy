Resource barriers and lifetimes are derived from RDG
RDG는 순서 해소기가 아니라 자원 의존성 해소기
RDG가 분석하는 건 FRDGBuilder::AddPass에 선언된 자원 read/write 의존성

AddPass는 그 패스의 pass parameter struct(=각 자원을 read 하는지 write 하는지 선언) 를 함께 받습니다

FRDGBuilder::Execute가 선언된 read/write로부터 자동으로

자원 배리어/전이 삽입 — 패스 B가 패스 A가 쓴 걸 읽으면, 
그 사이에 올바른 전이(RenderTarget→SRV 등)를 넣습니다

RDG 이전엔 RHICmdList.Transition(...)을 수동으로 삽입했습니다

---

각 자원의 (첫 write~마지막 read) 수명을 알므로, 
수명이 안 겹치는 자원끼리 같은 GPU 메모리를 공유시킵니다

RDGBuilder Execute는 예약된 순서 슬롯을 병렬로 채울 뿐
PrePass=슬롯0, ShadowDepth=슬롯1, BasePass=슬롯2 …

RenderViewFamily_RenderThread(GraphBuilder:FRDGBuilder)        
{
    FDeferredShadingSceneRenderer::Render(GraphBuilder);       // ① 그래프 빌드
    GraphBuilder:FRDGBuilder::Execute();                       // ② 그래프 실행
}

FDeferredShadingSceneRenderer::Render(GraphBuilder:FRDGBuilder)   
{
    FSceneRenderer::OnRenderBegin();                             // 씬 상태 커밋

    visTasks :IVisibilityTaskData = LaunchVisibilityTasks();      // ★비동기 발사 (자유 함수)

    FDeferredShadingSceneRenderer::BeginInitViews(visTasks);     // await + command 굽기 발사
    FDeferredShadingSceneRenderer::EndInitViews();               // 뷰 유니폼 버퍼

    // ── RDG 패스 등록만
    FDeferredShadingSceneRenderer::RenderPrePass();              // GraphBuilder.AddPass(...)
    FDeferredShadingSceneRenderer::RenderShadowDepthMaps();
    FDeferredShadingSceneRenderer::RenderBasePass();
    FDeferredShadingSceneRenderer::RenderLights();
    FDeferredShadingSceneRenderer::RenderTranslucency();
    // 그래프 완성
}

# ViewFamily를 렌더 스레드 작업으로 포장하고, FSceneRenderer들을 만든 뒤 SceneRenderBuilder에 등록한다.
FRendererModule::BeginRenderingViewFamily()
{
  World->SendAllEndOfFrameUpdates();

  Canvas->Flush_GameThread();
  # Canvas에 쌓인 draw item을 render command로 flush한다. ViewFamily 렌더링 전에 2D/Canvas 작업을 비운다.

  SceneRenderBuilder = FSceneRenderBuilder(Scene);

  SceneRenderBuilder.Execute();

}

# Deferred renderer의 한 프레임 graph를 조립하는 중심 함수.
# InitViews로 "무엇을 그릴지"를 확정하고, 이후 RDG pass들을 등록한다.
# 씬 커밋 → 가시성 → 뷰 init → RDG 패스 등록
FDeferredShadingSceneRenderer::Render(GraphBuilder, SceneUpdateInputs)
{
  InitViewTaskDatas = OnRenderBegin(GraphBuilder, SceneUpdateInputs);
  # scene update를 커밋하고 visibility task를 발사한다.

  BeginUpdateExposureCompensationCurveLUT();
  BeginVirtualTextureUpdate();
  PrepareFrameSystems();                          
  # Lumen, VSM, atmosphere, ray tracing, light function atlas 등 frame-level 준비.

  FSceneTextures::InitializeViewFamily(GraphBuilder, ViewFamily, FamilySize);
  # GBuffer/SceneDepth 등 씬 텍스처 RDG 할당

  BeginInitViews(GraphBuilder, ..., InitViewTaskDatas);
  // 가시성 결과 소비 시작 + 그림자/뷰 자원 준비

  PrepareMoreFrameResources();

  InitViewTaskDatas.VisibilityTaskData->FinishGatherDynamicMeshElements(...);
  # dynamic mesh gather 완료를 기다리고 mesh pass setup task를 시작한다.

  UpdateGPUSceneForViews(GraphBuilder);
  # view별 dynamic primitive data를 GPUScene에 업로드

  InstanceCullingManager.BeginDeferredCulling(GraphBuilder);
  // GPU 인스턴스 컬링 시작

  EndInitViews(GraphBuilder, ..., InitViewTaskDatas);
  # visibility/relevance/mesh pass setup 을 기다린다

  AddMainRenderPassesToRDG(GraphBuilder);
  # Depth/Base/Shadow/Lighting/Translucency 등 RDG pass들을 등록한다.

  AddPostProcessingAndFinalPassesToRDG(GraphBuilder);
  # post process, upscaler, final output 관련 pass들을 등록한다.
}

# Scene update와 visibility pipeline을 연결하는 함수.
# FScene의 primitive update를 커밋하고, 그 뒤 visibility task를 시작한다.
FSceneRenderer::OnRenderBegin()
{
  PrepareSceneTexturesConfigTask = launch async
  {
    for each Renderer
    {
      Renderer->PrepareViewRectsForRendering();
      // 각 뷰의 ViewRect(픽셀 영역) 확정
      InitializeSceneTexturesConfig(...);
      // 해상도/포맷 등 씬 텍스처 설정
    }
  }

  // 정적 메시 갱신이 끝난 뒤 실행될 콜백 등록
  SceneUpdateParameters.Callbacks.PostStaticMeshUpdate =
    lambda(StaticMeshUpdateTask)
    {
      await PrepareSceneTexturesConfigTask;

      ViewExtensions.PreRenderViewFamily_RenderThread();
      ViewExtensions.PreRenderView_RenderThread();

      PrepareViewStateForVisibility();

      VisibilityTaskData = LaunchVisibilityTasks(GraphBuilder.RHICmdList, StaticMeshUpdateTask);
      # frustum/occlusion/relevance/gather 계열 task를 발사한다.

      GPUSceneUpdateTaskPrerequisites.AddPrerequisites(VisibilityTaskData->GetComputeRelevanceTask());
      # GPUScene update가 relevance 계산 이후로 이어지도록 prerequisite를 건다.

      GPUSceneUpdateTaskPrerequisites.Trigger();
      # prerequisite event를 열어 GPUScene update 쪽 task가 진행 가능하게 한다
    };

  if (SceneUpdateInputs)
  {
    Scene->Update(GraphBuilder, SceneUpdateParameters);
  }
  else
  {
    PostStaticMeshUpdate({});
  }

  return VisibilityTaskData;
}

{
  for each View

  Tasks.DynamicMeshElementsPrerequisites.AddPrerequisites(Scene.GetGPUSkinUpdateTask());

  DynamicMeshElements.ContextContainer.Init(...);
  DynamicMeshElements.ViewCommandsPerView.SetNum(Views.Num());

  launch LightVisibility / FrustumCull / OcclusionCull / ComputeRelevance / Gather tasks;
}

# RDG 그래프를 실제 RHI command recording으로 컴파일/실행한다.
FRDGBuilder::Execute()
{
  FlushAccessModeQueue();

  EpiloguePass = AddGraphEpiloguePass();
  # graph 끝 sentinel pass를 추가한다. 외부 resource final state 정리에 필요하다.

  bCompiling = true;
  # 이제 AddPass 단계가 아니라 compile 단계로 들어간다.

  if not ImmediateMode
  {
    BeginFlushResourcesRHI();

    await ParallelSetupTasks at Compile point;
    await AsyncSetupQueue;

    ProcessAsyncSetupQueue();

    Compile();
      // pass culling
      // resource lifetime
      // barrier planning
      // transient allocation/deallocation
      // async compute fences
      // parallel execute setup
  }

  bCompiling = false;
  # 이제 실행 단계.

  if ParallelExecute enabled
  {
    launch SetupDispatchPassExecute task if needed;
    wake parallel execute tasks;
  }
  else
  {
    SetupDispatchPassExecute();
  }

  QueuedCmdLists = [];

  for each Pass in compiled RDG pass order
  {
    if Pass.bCulled
      continue;

    if Pass.bParallelExecute
    {
      if Pass is DispatchPass
      {
        await Pass.CommandListsEvent;
        QueuedCmdLists.Append(Pass.CommandLists);
      }
      else if Pass.bParallelExecuteBegin
      {
        QueuedCmdLists.Add(ParallelPassSet.CmdList);
      }

      if Pass.bDispatchAfterExecute
      {
        FlushParallel();
        RHICmdList.ImmediateFlush(DispatchToRHIThread);
      }
    }
    else
    {
      FlushParallel();

      ExecuteSerialPass(RHICmdList, Pass);

      if Pass.bDispatchAfterExecute
      {
        RHICmdList.ImmediateFlush(DispatchToRHIThread);
      }
    }
  }

  FlushParallel();

  await ParallelExecute.TasksAwait;

  ExtractExternalResources();
  SetEpilogueResourceAccesses();
}

FRendererModule::BeginRenderingViewFamily

게임 스레드에서 호출
“이번 ViewFamily를 렌더 스레드에서 렌더링해라”라는 작업 패키지를 만듭니다

World->SendAllEndOfFrameUpdates()로 컴포넌트들이 프레임 끝에 미뤄둔 렌더 상태 갱신 요청을 밀어냅니다. 
여기서 MarkRenderTransformDirty, MarkRenderDynamicDataDirty, MarkRenderStateDirty 같은 요청들이 렌더링 쪽으로 넘어갈 준비

아직 FScene의 최종 SoA 배열이 완전히 갱신됐다는 뜻은 아닙니다
“게임 스레드가 요청을 제출한 상태”

FSceneRenderBuilder를 만듭니다
이 builder는 “이번 scene 렌더링에 필요한 renderer들과 cleanup command들을 모아두는 포장지”

CreateLinkedSceneRenderers는 FSceneViewFamily를 기반으로 FSceneRenderer를 만듭니다
이 과정에서 FSceneView는 렌더러 내부용 확장 view인 FViewInfo로 복사/확장됩니다.

완성되는 정보

SceneRenderBuilder_Render / Renderer:FDeferredShadingSceneRenderer
SceneRenderBuilder_Render / Renderer::Views[i]:FViewInfo
SceneRenderBuilder_Render / Renderer::ViewFamily:FSceneViewFamily

LaunchVisibilityTasks

1

“visibility pipeline의 task graph를 구성하고 발사한 뒤, 핸들을 돌려주는 함수”입니다.

VisibilityTaskData / ViewPackets[i]:FVisibilityViewPacket
VisibilityTaskData / ViewPackets[i]::View:FViewInfo

frustum culling, occlusion culling, relevance 계산, dynamic mesh gather 을 작업하고
