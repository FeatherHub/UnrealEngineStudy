FSlateApplication::OnKeyDown(const FKeyEvent& InKeyEvent) {
    // 1. UI 위젯이 먼저 입력을 처리할 기회
    if (ProcessKeyDownEvent(InKeyEvent)) {
        return; // UI가 소비함
    }
    
    // 2. 게임 뷰포트로 전달
    if (GameViewport.IsValid()) {
        GameViewport->OnKeyDown(InKeyEvent);
    }
}

-------------------      --------------------------------------
::SetupInputComponent    ::Move

void AMyPlayerController::Move(const FInputActionValue& Value) {
    // 1. 입력 해석
    FVector2D MovementVector = Value.Get<FVector2D>();
    
    // 2. Pawn에게 명령
    if (APawn* Pawn = GetPawn()) {
        FVector Forward = GetControlRotation().Vector();
        FVector Right = FRotationMatrix(GetControlRotation()).GetScaledAxis(EAxis::Y);
        
        Pawn->AddMovementInput(Forward, MovementVector.Y);
        Pawn->AddMovementInput(Right, MovementVector.X);
    }
}
